Editor Class Example

#if UNITY_EDITOR
[CustomEditor(typeof(TargetClass))]
public class TargetClassEditor : Editor
{
    TargetClass script;

    private void OnEnable()
    {
        script = (TargetClass)target;
    }

    public override void OnInspectorGUI()
    {
        base.OnInspectorGUI();
        GUILayout.Space(10);

        if (GUILayout.Button("TargetClass özelliği"))
        {
            script.Function();
        }     
    }
}
#endif
